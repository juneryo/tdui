TdAccordion 折叠面板

这里是面板0
这里是面板1
这里是面板2
这里是面板3

示例代码：

HTML:
<td:accordion config="$opt" active="1">
	<div slot="content0">这里是面板0</div>
	<div slot="content1">这里是面板1</div>
	<div slot="content2">这里是面板2</div>
	<div slot="content3">这里是面板3</div>
</td:accordion>


JS:
require(['ui/accordion/td.accordion'], function () {
	var v = avalon.define({
		$id: 'test',
		$opt: {
			onchanged: function(ev, vm) {
				avalon.log('当前面板索引:' + vm.active);
			},
			panels: [{
				title: '面板0', fun: function(ev, vm) {
					//切换到该标签时触发
					avalon.log('切换到面板0');
				}
			}, {
				title: '面板1'
			}, {
				title: '面板2'
			}, {
				title: '面板3', fun: function(ev, vm) {
					avalon.log('展现面板3');
				}
			}]
		}
	});
	avalon.scan();
});
