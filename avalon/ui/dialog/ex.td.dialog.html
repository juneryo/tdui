示例代码：

HTML:
<td:dialog $id="dialog1" config="$opt" title="弹出框" size="large">
	<div slot="content">
		<a href="javascript:void(0)" ms-click="showDialog($event, 2)">普通弹出框</a>
	</div>
</td:dialog>
<td:dialog $id="dialog2" title="普通弹出框">
	<div slot="content">
		<a href="javascript:void(0)" ms-click="showDialog($event, 3)">小弹出框</a>
	</div>
</td:dialog>
<td:dialog $id="dialog3" title="小弹出框" size="small">
	<div slot="content">
		这是小弹出框
	</div>
</td:dialog>


JS:
require(['ui/dialog/td.dialog'], function () {
	var v = avalon.define({
		$id: 'test',
		showDialog: function(ev, n) {
			avalon.vmodels['dialog' + n].showDialog();
		},
		$opt: {
			onshowed: function(ev, vm) {
				avalon.log('dialog show');
			},
			onhided: function(ev, vm) {
				avalon.log('dialog hide');
			},
			buttons: [{display: '主按钮', type: 'primary', fun: function(ev, vm) {
					
			}}, {display: '成功按钮', type: 'success', fun: function(ev, vm) {
				
			}}, {display: '信息按钮', type: 'info', fun: function(ev, vm) {
				
			}}, {display: '警告按钮', type: 'warning', fun: function(ev, vm) {
				
			}}, {display: '危险按钮', type: 'danger', fun: function(ev, vm) {
				
			}}, {display: '关闭', fun: function(ev, vm) {
				vm.hideDialog();
			}}]
		}
	});
	avalon.scan();
});
