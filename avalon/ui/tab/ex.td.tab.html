示例代码：

HTML:
<td:tab config="$opt" active="3">
	<div slot="content0">标签0</div>
	<div slot="content1">标签1</div>
	<div slot="content2">标签2</div>
	<div slot="content3">标签3</div>
</td:tab>


JS:
require(['ui/tab/td.tab'], function () {
	var v = avalon.define({
		$id: 'test',
		$opt: {
			onchanged: function(ev, vm) {
				avalon.log('当前标签索引:' + vm.active);
			},
			tabs: [{
				title: '标签0', fun: function(ev, vm) {
					//切换到该标签时触发
					avalon.log('切换到标签0');
				}
			}, {
				title: '标签1'
			}, {
				title: '标签2'
			}, {
				title: '标签3', fun: function(ev, vm) {
					avalon.log('展现标签3');
				}
			}]
		}
	});
	avalon.scan();
});
