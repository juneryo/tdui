TdTree 树

示例代码：

HTML:
<td:tree config="$opt" checkbox="true"></td:tree>


JS:
require(['ui/tree/td.tree'], function () {
	var dat = [
		{text: "菜单01", expand:true, checked:false, children: [
			{text: '菜单01-01', expand:false, checked:false, children: []},
			{text: 2222, expand:true, checked:false, children: [
				{text: 777, expand:false, checked:false, children: []}
			]},
			{text: 3333, expand:true, checked:false, children: [
				{text: 8888, expand:false, checked:false, children: []},
				{text: 999, expand:false, checked:false, children: []}
			]}
		]},
		{text: "菜单02", expand:true, checked:false, children: [
			{text: 4444, expand:false, checked:false, children: []},
			{text: 5555, expand:false, checked:false, children: []},
			{text: 6666, expand:false, checked:false, children: []}
		]},
		{text: "ccc", expand:false, checked:false, children: []},
		{text: "ddd", expand:false, checked:false, children: []},
		{text: "eee", expand:false, checked:false, children: []},
		{text: "fff", expand:false, checked:false, children: []}
	];
	var v = avalon.define({
		$id: 'test',
		$opt: {
			onclicked: function(el, vm) {
				avalon.log(el);
			},
			oncollapsed: function(el, vm) {
				avalon.log('折叠面板' + el.text);
			},
			onexpanded: function(el, vm) {
				avalon.log('展开面板' + el.text);
			},
			data: dat
		}
	});
	avalon.scan();
});
