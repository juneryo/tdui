TdButton 按钮

Action Button:

Button:

示例代码：

HTML:
<td:button config="$opt"  value="禁用default action按钮" disabled="true"></td:button>
<td:button config="$opt1" value="primary按钮" badge="5" type="primary"></td:button>
<td:button config="$opt2" value="大success按钮" type="success" size="large"></td:button>
<td:button config="$opt3" value="普通info按钮" type="info"></td:button>
<td:button config="$opt4" value="小warning按钮" badge="5" type="warning" size="small"></td:button>
<td:button config="$opt5" value="超小禁用danger按钮" type="danger" size="xsmall" disabled="true"></td:button>


JS:
require(["ui/button/td.button"], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			click: function(ev, vm) {
				alert('default');
			},
			actions: [{display: '操作1', fun: function(ev, vm) {
				alert('操作1');
			}},{display: '操作2', fun: function(ev, vm) {
				alert('操作2');
			}}]
		},
		$opt1: {
			click: function(ev, vm) {
				alert('primary');
			},
			actions: [{display: '操作1', fun: function(ev, vm) {
				alert('primary操作1');
			}},{display: '操作2', fun: function(ev, vm) {
				alert('primary操作2');
			}}, {type:'split'}, {display: '操作3', fun: function(ev, vm) {
				alert('primary操作3');
			}}]
		},
		$opt2: {
			click: function(ev, vm) {
				alert('success');
			}
		},
		$opt3: {
			click: function(ev, vm) {
				alert('info');
			}
		},
		$opt4: {
			click: function(ev, vm) {
				alert('warning');
			}
		},
		$opt5: {
			click: function(ev, vm) {
				alert('danger');
			}
		}
	});
	avalon.scan();
});
