TdCheckbox 选择框



示例代码：

HTML:
<td:checkbox config="$opt" checked="true" disabled="false"></td:checkbox>
<td:checkboxgroup config="$group_opt" label="选择" name="group"></td:checkboxgroup>


JS:
require(['ui/form/checkbox/td.checkbox', 'ui/form/checkbox/td.checkboxgroup'], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			
		},
		$group_opt: {
			checkboxes:[
				{label:'升职', checked:true , disabled:false, value:'1'},
				{label:'加薪', checked:false, disabled:false, value:'2'},
				{label:'减薪', checked:false, disabled:true,  value:'3'}
			]
		}
	});
	avalon.scan();
});
