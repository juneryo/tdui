TdCheckbox 选择框

示例代码：

HTML:
<td:checkbox config="$opt" checked="true" disabled="false"></td:checkbox>


JS:
require(['ui/form/checkbox/td.checkbox'], function () {
    avalon.define({
        $id: 'test',
        $opt: {
            //在td:checkbox标签中不定义属性，也可以在这里定义。若标签中和$opt中都不定义则使用默认。
            //label:"Checkbox后面的名称"
            //checked:true|false
            //disabled:true|false
            //标签中属性优先级最高、其次$opt中、再次默认值
            //注意：标签中属性命名方式不要以驼峰式命名，要以横线分割，例：labelText="Boy"(错) label-text="Boy"(对) 
        }
    });
    avalon.scan();
});
