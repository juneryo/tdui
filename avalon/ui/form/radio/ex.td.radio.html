示例代码：

HTML:
<td:radio config="$opt" label="Boy" checked="true" disabled="false"></td:radio>


JS:
require(['ui/form/radio/td.radio', 'ui/form/radio/td.radiogroup'], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			onchecked: function(ev, vm) {
				avalon.log('选中事件 checked: ' + vm.checked);
			},
			onchanged: function(ev, vm) {
				avalon.log('变更事件 checked: ' + vm.checked);
			}
		},
		$group_opt: {
			onchanged: function(ev, vm, radio) {
				avalon.log('checked radio is changed, radio [' + radio.label + '] is checked.');
			},
			radios: [
				{label:'博士', checked:true , disabled:false, value:'1', onchecked: function(ev, vm, radio) {
					avalon.log('博士radio checked:');
					avalon.log(vm.getData());
				}},
				{label:'硕士', checked:false, disabled:false, value:'2', onchanged: function(ev, vm, radio) {
					avalon.log('硕士radio changed:' + vm.value);
					if(radio.checked == true) {
						vm.radios[2].disabled = false;
					}else {
						vm.radios[2].disabled = true;
					}
				}},
				{label:'本科', checked:false, disabled:true, value:'3', onchanged: function(ev, vm, radio) {
					avalon.log('本科radio changed:');
					avalon.log(vm.getData());
				}},
				{label:'大专', checked:false, disabled:true, value:'4'}
			]
		}
	});
	avalon.scan();
});
