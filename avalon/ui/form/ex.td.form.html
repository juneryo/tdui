TdForm 表单

示例代码：

HTML:
<td:form $id="ex-form" config="$opt" title="用户注册">
	<div slot="content">
		<div class="row">
			<td:text label="姓名" name="name" must="true" class="col-xs-12 col-sm-12"></td:text>
		</div>
		<div class="row">
			<td:spinner label="年龄" name="age" min="1" max="100" class="col-xs-12 col-sm-6"></td:spinner>
		</div>
		<div class="row">
			<td:select config="$select_opt" label="性别" name="sex" class="col-xs-12 col-sm-6"></td:select>
		</div>
		<div class="row">
			<td:switch config="$switch_opt" label="状态" name="status" class="col-xs-12 col-sm-12"></td:switch>
		</div>
		<div class="row">
			<td:datepicker label="出生日期" name="date" must="true" class="col-xs-12 col-sm-6"></td:datepicker>
		</div>
		<div class="row">
			<td:radiogroup config="$radiogroup_opt" label="学历" name="edu" class="col-xs-12 col-sm-12"></td:radiogroup>
		</div>
		<div class="row">
			<td:checkboxgroup config="$checkboxgroup_opt" label="爱好" name="favo" class="col-xs-12 col-sm-12"></td:checkboxgroup>
		</div>
		<div class="row">
			<td:rate label="信用" name="credit" half="true" num="5" value="2.5" class="col-xs-12 col-sm-12"></td:rate>
		</div>
		<div class="row">
			<td:textarea label="备注" name="comment" class="col-xs-12 col-sm-12"></td:textarea>
		</div>
	</div>
</td:form>


JS:
require(['ui/form/td.form', 'ui/form/text/td.text', 'ui/form/select/td.select',
	'ui/form/spinner/td.spinner', 'ui/form/switch/td.switch', 'ui/form/rate/td.rate', 'ui/form/datepicker/td.datepicker',
	'ui/form/radio/td.radiogroup', 'ui/form/checkbox/td.checkboxgroup', 'ui/form/textarea/td.textarea'], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			submitCallback: function(ev, vm, data) {
				alert('submit');
			},
			resetCallback: function(ev, vm) {
				alert('reset');
			}
		},
		$select_opt: {
			data: {
				'1': '男',
				'2': '女',
				'3': '其他'
			},
		},
		$switch_opt: {
			display: {
				on: '男',
				off: '女'
			}
		},
		$radiogroup_opt: {
			radios: [
				{label:'大专', checked:false, disabled:false, value:'1'},
				{label:'本科', checked:true, disabled:false, value:'2'},
				{label:'硕士', checked:false, disabled:false, value:'3'},
				{label:'博士', checked:false, disabled:false, value:'4'}
			]
		},
		$checkboxgroup_opt: {
			checkboxes:[
				{label:'电影', checked:false, disabled:false, value:'1'},
				{label:'音乐', checked:false, disabled:false, value:'2'},
				{label:'游戏', checked:false, disabled:false, value:'3'},
				{label:'旅游', checked:false, disabled:false, value:'4'}
			]
		}
	});
	avalon.scan();
	
	avalon.vmodels['ex-form'].setData({
		name: 'yujun',
		age: '30',
		sex: '1',
		edu: '3',
		favo: '1,2,3,4',
		credit: '5',
		comment: 'i am yj'
	});
});
