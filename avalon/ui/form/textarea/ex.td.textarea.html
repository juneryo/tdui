TdTextarea 多行输入框

示例代码：

HTML:
<!--
	校验属性 must(是否必选): true/false(default), html(是否支持输入html): true/false(default)
-->
<td:textarea config="$opt" label="备注" name="mark" must="true" maxlen="200" value="这里是备注" width="50%"></td:textarea>


JS:
require(['ui/form/textarea/td.textarea'], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			onclicked: function(ev, vm) {
				avalon.log(vm.getData());
			},
			onchanged: function(data, vm) {
				avalon.log(data);
			}
		}
	});
	avalon.scan();
});
