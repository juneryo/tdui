TdSpinner 加减输入框

示例代码：

HTML:
<td:spinner config="$opt" label="年龄" name="age" disabled="false" value="21" max="99" min="1"></td:spinner>


JS:
require(['ui/form/spinner/td.spinner'], function () {
	var vm = avalon.define({
		$id: 'test',
		$opt: {
			onchanged: function(ev, vm) {
				avalon.log('value changed to:' + vm.value);
			},
			onupclicked: function(ev, vm) {
				avalon.log('click up');
			},
			ondownclicked: function(ev, vm) {
				avalon.log('click down');
			}
		}
	});
	avalon.scan();
});
