TdSelect 下拉框

示例代码：

HTML:
<td:select config="$opt" label="国家" name="contry" disabled="false" muti="true"></td:select>


JS:
require(['ui/form/select/td.select'], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			/* 响应数据格式：
			 * {"rspcod": "200", "data": {
			 *   "key1": "val1", "key2": "val2"
			 * }}
			 */
			//url: 'xxx',  //与属性 data 不能共存
			data: {
				'china': '中国',
				'japan': '日本',
				'korea': '韩国',
				'america': '美国',
				'england': '英格兰',
				'france': '法国'
			},
			selected: ['china', 'japan'],
			onselected: function(ev, vm) {
				avalon.log('选中[' + vm.selectedValue + ':' + vm.data[vm.selectedValue] + ']');
			},
			onchanged: function(ev, vm) {
				avalon.log(vm.getData());
			},
			onloaded: function(data, vm) {
				avalon.log('ajax load data completed');
				avalon.log(data);
			}
		}
	});
	avalon.scan();
});
