TdCheckboxGroup 选择框组

示例代码：

HTML:
<td:checkboxgroup config="$opt"></td:checkboxgroup>


JS:
require(['ui/form/checkboxgroup/td.checkboxgroup'], function () {
    avalon.define({
        $id: 'test',
        $opt: {
            //在td:checkboxgroup标签中不定义属性，也可以在这里定义。若标签中和$opt中都不定义则使用默认。
            //标签中属性优先级最高、其次$opt中、再次默认值
            //group:"checkbox组"
                //name:"Checkbox的标识"
                //label:"中文名称"
                //childs:具体内容
            group:{
                name:"testGroupCheckbox",
                label:"你想要：",
                childs:[
                    {label:"升职",checked:true ,disabled:false,value:"1"},
                    {label:"加薪",checked:false,disabled:false,value:"2"}
                ]}
        }
    });
    avalon.scan();
});
