TdCombobox 下拉框

示例代码：

HTML:
<td:combobox config="$opt" checked="true" disabled="false"></td:combobox>


JS:
require(['ui/form/combobox/td.combobox'], function () {
    avalon.define({
        $id: 'test',
        $opt: {
            label        : "理想的女票:",//下拉中文名称
            defaultOption: "--请选择--", //未选择时的默认显示中文内容
            isMutiSelect : true,         //是否可多选
            isShowUl     : false,        //显示UL状态
            disabled     : false,        //是否可用
            //默认下拉内容
            group        : [
                            {text:"--请选择--",value:"",checked:false},
                            {text:"凤姐",value:"1",checked:false},
                            {text:"芙蓉",value:"2",checked:false},
                            {text:"如花",value:"3",checked:false},
                            {text:"美美",value:"4",checked:false}
                            ]
        }
    });
    avalon.scan();
});
