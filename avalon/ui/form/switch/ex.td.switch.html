TdSwitch 开关

示例代码：

HTML:
<td:switch config="$opt" label="状态" name="status" disabled="false" on="true"></td:switch>


JS:
require(['ui/form/switch/td.switch'], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			display: {
				on: '可用',
				off: '禁用'
			},
			onclicked: function(ev, vm) {
				avalon.log('switch clicked');
			},
			onchanged: function(ev, vm) {
				avalon.log('switch changed to ' + (vm.on == true ? 'on' : 'off'));
			}
		}
	});
	avalon.scan();
});
