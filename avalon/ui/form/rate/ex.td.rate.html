TDRate 评分

示例代码：

HTML:
<td:rate config="$opt" label="评分" name="rate" disabled="false" half="true" num="5" default="3.5"></td:rate>
			

JS:
require(['ui/form/rate/td.rate'], function () {
	avalon.define({
		$id: 'test',
		$opt: {
			onclicked: function(ev, vm) {
				avalon.log(vm.getData());
			},
			onchanged: function(ev, vm) {
				avalon.log('value changed to: ' + vm.value);
			}
		}
	});
	avalon.scan();
});
