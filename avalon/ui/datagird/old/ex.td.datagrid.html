TdDataGrid 数据表格

示例代码：

	var cols = [
		{name: 'id', display: '编号', width: 100, type: 'text', value: '', row: 1, col: 3, disabled: true}, 
		{name: 'name', display: '名称', width: 150, type: 'text', value: '', row: 1, col: 3}, 
		{name: 'price', display: '价格', width: 120, type: 'text', value: '', row: 1, col: 3},
		{name: 'category', display: '类别', width: 150, type: 'select', value: '', option: {
			'A1': '游戏', 'A2' : '电影', 'A3' : '音乐'
		}, row: 1, col: 3}
	];
	require(["ui/datagird/td.datagrid"], function () {
		avalon.define({
			$id: 'test',
			$opt: {
				title: '测试数据表格',
				h: 300,              //表体高度(可选)
				loadUrl: '/tdui/data/td.datagrid.json',
				cols: cols,
				checkbox: true,      //是否显示checkbox(默认true)
				singleSelect: false, //是否单选(默认false)
				operations: [{display: '新增', icon: 'glyphicon glyphicon-plus', fun: function(ev, vm) {
					alert('新增');
				}}, {display: '删除', icon: 'glyphicon glyphicon-remove', fun: function(ev, vm) {
					alert('删除');
				}}]
			}
		});
		avalon.scan();
	});
