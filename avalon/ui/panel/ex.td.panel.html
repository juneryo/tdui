TdPanel 面板
{{submitInfo}}

示例代码：

HTML:
<td:panel config="$opt" title="注册面板">
	<div slot="content">
		<td:form config="$form_opt" title="用户注册" class="col-xs-10 col-sm-8 col-xs-offset-1 col-sm-offset-2">
			<div slot="content">
				<div class="row">
					<td:text label="用户名" name="name" must="true" class="col-xs-12 col-sm-6"></td:text>
				</div>
				<div class="row">
					<td:radiogroup config="$sex_opt" label="性别" name="sex" class="col-xs-12 col-sm-12"></td:radiogroup>
				</div>
				<div class="row">
					<td:datepicker label="出生日期" name="date" must="true" class="col-xs-12 col-sm-6"></td:datepicker>
				</div>
				<div class="row">
					<td:text label="邮件地址" name="email" valid="email" must="true" class="col-xs-12 col-sm-6"></td:text>
				</div>
				<div class="row">
					<td:text label="手机号码" name="mobile" valid="mobile" must="true" class="col-xs-12 col-sm-6"></td:text>
				</div>
			</div>
		</td:form>
	</div>
	<div slot="footer">
		{{submitInfo}}
	</div>
</td:panel>


JS:
require(['ui/panel/td.panel', 'ui/form/td.form', 'ui/form/text/td.text', 
	'ui/form/radio/td.radiogroup', 'ui/form/datepicker/td.datepicker'], function () {
	var v = avalon.define({
		$id: 'test',
		submitInfo: '',
		$opt: {
			buttons: [{
				display: '帮助', icon: 'glyphicon glyphicon-question-sign', fun: function(ev, vm) {
					alert('帮助');
				}
			}]
		},
		$form_opt: {
			resetCallback: function(ev, vm) {
				v.submitInfo = vm.getData();
			}
		},
		$sex_opt: {
			radios: [
				{label:'男', checked:true, disabled:false, value:'1'},
				{label:'女', checked:false, disabled:false, value:'2'}
			]
		}
	});
	avalon.scan();
});
