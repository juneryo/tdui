高级按钮

带有其他操作项的按钮
<td:button config="$opt" value="禁用default action按钮" disabled="true"></td:button>
<td:button config="$opt" value="带徽章primary action按钮" badge="7" type="primary"></td:button>	
			
var vbutton = avalon.define({
	$id: 'button02',
	$opt: {
		onclicked: function(ev, vm) {
			TD.alert('default');
		},
		actions: [{display: '操作1', fun: function(ev, vm) {
			TD.alert('primary操作1');
		}},{display: '操作2', fun: function(ev, vm) {
			TD.alert('primary操作2');
		}}, {
			type: 'split'
		}, {
			display: '操作3', fun: function(ev, vm) {
			TD.alert('primary操作3');
		}}]
	}
});
